1 Proteome & transcriptome comparison

We start comparing the dynamics of genes to matching proteins.

Protein values range 0-10 while using multiplex TMTpro while we normalize the counts and/or and take the log2 of the TPM since the amplitude of the RNA counts varies greatly across multiple orders of magnitude.

calculate_correlations <- function(data) {
  num_columns <- ncol(data) # 16
  half_columns <- num_columns / 2 # 8 rna + 8 protein
  
  correlations <- data.frame(gene_protein_Id = rownames(data), 
                             correlation = numeric(nrow(data))) # create an empty dataframe
  for (i in 1:nrow(data)) {
    rna_values <- as.numeric(data[i, 1:half_columns]) # get vector rna 
    prot_values <- as.numeric(data[i, (half_columns + 1):num_columns]) # get vector protein 
    
    correlations[i, "correlation"] <- cor(rna_values, prot_values, method = "pearson")
  }
  
  return(correlations)
}

compute_multiple_correlation <- function(data) {
  num_genes <- nrow(data)
  half_columns <- ncol(data) / 2
  
  results <- list()
  
  for (i in 1:num_genes) {
    
    #split the data half rna and prot
    rna_values <- as.numeric(data[i, 1:half_columns])
    prot_values <- as.numeric(data[i, (half_columns+1):ncol(data)])
    
    # calculate similarity | distance 
    pearson <- cor(rna_values, prot_values, method = "pearson")
    spearman <- cor(rna_values, prot_values, method = "spearman")
    cosine <- cosine(rna_values, prot_values)

    results[[rownames(data)[i]]] <- list(
      rna_values = rna_values,
      prot_values = prot_values,
      pearson_correlation = pearson,
      spearman_correlation = spearman,
      cosine_similarity = cosine
    )
  }
  
  return(results)
}

select_top_bottom_rows <- function(df, column_name, n = 6) {
  top_rows <- df %>%
    arrange(desc({{ column_name }})) %>%
    head(n)
  
  bottom_rows <- df %>%
    arrange({{ column_name }}) %>%
    head(n)
  
  return(list(top = top_rows, bottom = bottom_rows) )
}

# Create a function to generate heatmaps
create_heatmap <- function(vector_name, rna_values, prot_values, cosine_coefficient, pearson_coefficient) {
  df <- data.frame(time = 1:8, RNA = rna_values, Protein = prot_values)
  df_long <- gather(df, key = "Type", value = "Value", -time)
  
  p <- ggplot(df_long, aes(x = time, y = Value, color = Type)) +
    geom_point(size = 3) +
    geom_line(size = 1) +
    labs(title = paste(vector_name),
         subtitle = paste("Cosine:", round(cosine_coefficient, 2), "Pearson:", round(pearson_coefficient, 2)),
         x = "Time",
         y = "Counts",
         color = "Type") +
    guides(color = guide_legend(order = 2)) +
    theme_pubr(legend = "bottom")
  
  return(p)
}

# Correct the mistake in generate_heatmaps function
generate_heatmaps <- function(top_bottom, metric_name) {
  top_heatmaps <- list()
  bottom_heatmaps <- list()
  
  for (i in 1:nrow(top_bottom$top)) {
    vector_name <- top_bottom$top$Gene_Name[i]
    vector_rna_values <- correlation_data[[vector_name]]$rna_values
    vector_prot_values <- correlation_data[[vector_name]]$prot_values
    cosine_coefficient <- top_bottom$top$cosine_similarity[i]
    pearson_coefficient <- top_bottom$top$pearson_correlation[i]

    heatmap <- create_heatmap(vector_name, vector_rna_values, vector_prot_values, cosine_coefficient, pearson_coefficient)
    bottom_heatmaps[[i]] <- heatmap
  }
  for (i in 1:nrow(top_bottom$bottom)) {
    vector_name <- top_bottom$bottom$Gene_Name[i]
    vector_rna_values <- correlation_data[[vector_name]]$rna_values
    vector_prot_values <- correlation_data[[vector_name]]$prot_values
    cosine_coefficient <- top_bottom$bottom$cosine_similarity[i]
    pearson_coefficient <- top_bottom$bottom$pearson_correlation[i]

    
    heatmap <- create_heatmap(vector_name, vector_rna_values, vector_prot_values, cosine_coefficient, pearson_coefficient)
    top_heatmaps[[i]] <- heatmap
  }
  
  # Combine the heatmaps
  top_row <- wrap_plots(top_heatmaps) + plot_layout(heights = c(1, 1)) + plot_annotation(title = paste("Distance:", metric_name))
  bottom_row <- wrap_plots(bottom_heatmaps) + plot_layout(heights = c(1, 1)) + plot_annotation(title = paste("Distance:", metric_name))
  combined_plot <- top_row / bottom_row
  
  return(combined_plot)
}
